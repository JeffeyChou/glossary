conjugateGradient[A_, b_, x0_, tol_, MaxIterations_] := 
 Module[{x, r, p, alpha, beta, i}, x = x0;
  r = b - A . x;
  p = r;
  i = 0;
  While[Norm[r] > tol && i < MaxIterations, 
   alpha = (r . r)/(p . A . p);
   x = x + alpha*p;
   rnew = r - alpha*A . p;
   beta = (rnew . rnew)/(r . r);
   p = rnew + beta*p;
   r = rnew;
   i = i + 1;];
  {x, i}]
A = {{4, -2, 0}, {-2, 2, -1}, {0, -1, 5}};
